#include <string.h> 
int main() { 
    int t; 
    scanf("%d", &t); // read the number of test cases 
    while (t--) { 
        char s1[10005], s2[10005], ans[20005]; 
        scanf("%s %s", s1, s2); // read the two strings 
        int len1 = strlen(s1), len2 = strlen(s2); 
        int i = 0, j = 0, k = 0; 
        while (i < len1 && j < len2) { 
            ans[k++] = s1[i++]; 
            ans[k++] = s2[j++]; 
        } 
        while (i < len1) ans[k++] = s1[i++]; 
        while (j < len2) ans[k++] = s2[j++]; 
        ans[k] = '\0'; 
        printf("%s\n", ans); 
    } 
    return 0; 
}
