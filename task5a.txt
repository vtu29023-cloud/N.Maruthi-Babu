#include <stdio.h> 
#include <math.h> 
#define MOD 1000000007 
int main() { 
    int t; 
    scanf("%d", &t); 
    while (t--) { 
        long long n; 
        scanf("%lld", &n); 
        long long min_sum = ((n - 1) * (n - 1) * n / 4) % MOD; // Minimum sum 
        long long max_sum = ((n * (n + 1)) * (2 * (n + 1)) / 12) % MOD; // Maximum sum 
        printf("%lld %lld\n", min_sum, max_sum); 
    } 
    return 0; 
}
