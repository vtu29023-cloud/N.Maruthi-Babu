#include <stdio.h>
#include <string.h>

int main() {
    char sentence[100];
    char replace_char;

    printf("Enter a sentence: ");
    fgets(sentence, 100, stdin);

    printf("Enter a character to replace: ");
    scanf("%c", &replace_char);

    for (int i = 0; i < strlen(sentence); i++) {
        if (sentence[i] == replace_char) {
            sentence[i] = '#';
        }
    }

    printf("Modified sentence: %s", sentence);

    return 0;
}
